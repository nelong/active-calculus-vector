\begin{align*} \int_{C_1} f_3 \, ds \amp = \int_{-\frac{\pi}{4}}^{\frac{3\pi}{4}}\left(\cos(t)-\sin(t)\right) (1) dt\\ \amp = \sin(\frac{3\pi}{4})-\cos(\frac{3\pi}{4}) - \left(\sin(-\frac{\pi}{4})-\cos(-\frac{\pi}{4})\right)=0 \end{align*}
