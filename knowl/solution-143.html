\begin{equation*} \mathbf{F}(\mathbf{r}(t)) = \left\lt 5 \sin ( t^3), -5 \cos (-2 t^2), -3 t^4 \right> \end{equation*}
\begin{equation*} \mathbf{r} '(t) = \left\lt 3 t^2, -4 t, -3 \right > \end{equation*}
\begin{equation*} \mathbf{F}(\mathbf{r}(t)) \cdot \mathbf{r} '(t) = \left\lt 5 \sin ( t^3), -5 \cos (-2 t^2), -3 t^4 \right> \cdot \left\lt 3 t^2, -4 t, -3 \right > \end{equation*}
\begin{equation*} =15 t^2 \sin ( t^3) + 20 t \cos (-2 t^2) + 9 t^4 \end{equation*}
The line integral is then:
\begin{equation*} \int_{\mathcal{C}} \mathbf{F} \cdot \,d\ \mathbf{r} = \int^1_0 \mathbf{F}(\mathbf{r}(t)) \cdot \mathbf{r} '(t) \ dt = \int^1_0 15 t^2 \sin ( t^3) + 20 t \cos (-2 t^2) + 9 t^4\ dt \end{equation*}
\begin{equation*} = \left [ -5 \cos ( t^3) - 5 \sin (-2 t^2) + \frac{9}{5} t^5 \right ]^1_0 \end{equation*}
\begin{equation*} = -5 \cos (1) - 5 \sin (-2) + \frac{34}{5} \end{equation*}
